          &gt;
            &lt;div&gt;
              {{ person.name }}
            &lt;/div&gt;

              [@rotate]="accordion.opened ? 'rotated' : 'none'"

              &lt;path stroke-linecap="round" stroke-linejoin="round" d="m19.5 8.25-7.5 7.5-7.5-7.5" /&gt;
            &lt;/svg&gt;

          &lt;ng-template headwindAccordionContent&gt;
            &lt;div [@expand] class="space-y-4 overflow-hidden py-3 text-sm"&gt;
              &lt;div class="flex items-center space-x-1"&gt;

                    {{ person.job }}

      state(
        'void',
        style({
          height: 0,
        }),
      ),
      transition(
        'void =&gt; *',
        animate(
          '.1s',
          style({
            height: '*',
          }),
        ),
      ),
      transition(
        '* =&gt; void',
        animate(
          '.1s',
          style({
            height: 0,
          }),
        ),
      ),
    ]),
    trigger('rotate', [
      state(
        'none',
        style({
          transform: 'rotate(0)',
        }),
      ),
      state(
        'rotated',
        style({
          transform: 'rotate(180deg)',
        }),
      ),
      transition('none &lt;=&gt; rotated', animate('.1s')),
    ]),
  ],
})

Accordion Classes

Basic Classes

Each component used to implement Accordion has the same class name as its name. You can set styles using the class name as a selector in a style file.

Button 
import { Component } from '@angular/core';
import {
  HeadwindAccordionButtonComponent,
  HeadwindAccordionComponent,
  HeadwindAccordionContentDirective,
} from '@favian/headwind-ui';

@Component({
  selector: 'app-accordion-classes-example',
  standalone: true,
  imports: [HeadwindAccordionComponent, HeadwindAccordionButtonComponent, HeadwindAccordionContentDirective],
  template: `
    &lt;headwind-accordion&gt;
      &lt;headwind-accordion-button&gt; Button &lt;/headwind-accordion-button&gt;

      &lt;ng-template headwindAccordionContent&gt;
        &lt;div class="mt-3"&gt;Content&lt;/div&gt;
      &lt;/ng-template&gt;
    &lt;/headwind-accordion&gt;
  `,
  styles: [
    `
      @tailwind components;

      @layer components {
        .headwind-accordion {
          @apply block w-full max-w-[400px] rounded-2xl bg-white p-8 shadow;
        }

        .headwind-accordion-button {
          @apply flex items-center justify-between rounded-md bg-rose-500 px-3 py-2 text-sm text-white transition-colors hover:bg-rose-400;
        }
      }
    `,
  ],
})
export class AccordionClassesExampleComponent {}

Opened State

You can use the opened attribute to change the display state of the content. When the opened state changes, the changed value is emitted through the openedChange emitter.

Button Opened: true
Content
import { Component } from '@angular/core';
import {
  HeadwindAccordionButtonComponent,
  HeadwindAccordionComponent,
  HeadwindAccordionContentDirective,
} from '@favian/headwind-ui';

@Component({
  selector: 'app-accordion-opened-state-example',
  standalone: true,
  imports: [HeadwindAccordionComponent, HeadwindAccordionButtonComponent, HeadwindAccordionContentDirective],
  template: `
    &lt;headwind-accordion
      [(opened)]="opened"
      class="block w-full max-w-[400px] space-y-2 rounded-2xl bg-white p-8 shadow"
    &gt;
      &lt;headwind-accordion-button
        class="block rounded-md bg-rose-500 px-3 py-2 text-sm text-white transition-colors hover:bg-rose-400"
      &gt;
        Button Opened: {{ opened }}
      &lt;/headwind-accordion-button&gt;

      &lt;ng-template headwindAccordionContent&gt;
        &lt;div class="mt-3"&gt;Content&lt;/div&gt;
      &lt;/ng-template&gt;
    &lt;/headwind-accordion&gt;
  `,
})
export class AccordionOpenedStateExampleComponent {
  opened = true;
}

toggle, open, close

The &lt;headwind-accordion&gt; component provides the toggle(), open(), and close() methods.

Button 
import { Component } from '@angular/core';
import {
  HeadwindAccordionButtonComponent,
  HeadwindAccordionComponent,
  HeadwindAccordionContentDirective,
} from '@favian/headwind-ui';

@Component({
  selector: 'app-accordion-methods-example',
  standalone: true,
  imports: [HeadwindAccordionComponent, HeadwindAccordionButtonComponent, HeadwindAccordionContentDirective],
  template: `
    &lt;div class="w-full max-w-[400px]"&gt;
      &lt;div class="flex flex-wrap items-stretch gap-2"&gt;
        &lt;button
          (click)="accordion.toggle()"
          class="min-h-8 flex-[1_1_0] rounded-md bg-rose-500 px-3 text-sm text-white"
        &gt;
          Toggle Accordion
        &lt;/button&gt;
        &lt;button (click)="accordion.open()" class="min-h-8 flex-[1_1_0] rounded-md bg-rose-500 px-3 text-sm text-white"&gt;
          Open Accordion
        &lt;/button&gt;
        &lt;button (click)="accordion.close()" class="min-h-8 flex-[1_1_0] rounded-md bg-rose-500 px-3 text-sm text-white"&gt;
          Close Accordion
        &lt;/button&gt;
      &lt;/div&gt;

      &lt;headwind-accordion #accordion class="mt-3 block space-y-2 rounded-2xl bg-white p-8 shadow"&gt;
        &lt;headwind-accordion-button
          class="block rounded-md bg-rose-500 px-3 py-2 text-sm text-white transition-colors hover:bg-rose-400"
        &gt;
          Button
        &lt;/headwind-accordion-button&gt;

        &lt;ng-template headwindAccordionContent&gt;
          &lt;div class="mt-3"&gt;Content&lt;/div&gt;
        &lt;/ng-template&gt;
      &lt;/headwind-accordion&gt;
    &lt;/div&gt;
  `,
})

Accessibility

Mouse Interaction

Clicking on the &lt;headwind-accordion-button&gt; element toggles its contents.

Keyboard Interaction

Command Target Description
Space, Enter &lt;headwind-accordion-button&gt; Toggle the contents.

API

HeadwindAccordionComponent

The root component of Accordion.

Selector

&lt;headwind-accordion&gt;

Host Class

.headwind-accordion<br/> .headwind-opened when Accordion opened.

Inputs

Name Description
@Input() set opened(value: boolean) Set the opened state of content.

Outputs

Name Description
@Output() openedChange: EventEmitter&lt;boolean&gt; Emits the changed opened state of content.

Methods

Name Description
toggle() Toggle the opened state of Content.
open() Open the content. If it is already open, it is ignored.
close() Close the content. If it is already closed, it is ignored.

HeadwindAccordionButtonComponent

A component that can open and close the Accordion content.

Selector

&lt;headwind-accordion-button&gt;

Host Class

.headwind-accordion-button

HeadwindAccordionContentDirective

A directive for a template that wraps Accordion content.

Selector

ng-template[headwindAccordionContent]
