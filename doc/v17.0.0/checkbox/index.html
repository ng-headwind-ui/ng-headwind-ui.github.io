Checkbox

Component that toggles boolean state.

import { Component } from '@angular/core';
import {
  HeadwindCheckboxButtonComponent,
  HeadwindCheckboxCheckedDirective,
  HeadwindCheckboxComponent,
} from '@favian/headwind-ui';
import { animate, state, style, transition, trigger } from '@angular/animations';

@Component({
  selector: 'app-basic-checkbox-example',
  standalone: true,
  imports: [HeadwindCheckboxComponent, HeadwindCheckboxButtonComponent, HeadwindCheckboxCheckedDirective],
  template: `
    &lt;headwind-checkbox #checkbox class="inline-flex cursor-pointer select-none items-center space-x-1"&gt;
      &lt;headwind-checkbox-button
        [class]="checkbox.checked ? 'border-rose-500 bg-rose-500' : 'bg-white'"
        class="relative flex h-5 w-5 items-center justify-center rounded border border-zinc-300"
      &gt;
        &lt;ng-template headwindCheckboxChecked&gt;
          &lt;svg
            [@scaleUp]
            xmlns="http://www.w3.org/2000/svg"
            fill="none"
            viewBox="0 0 24 24"
            stroke-width="1.5"
            stroke="currentColor"
            class="absolute h-4 w-4 stroke-white"
          &gt;
            &lt;path stroke-linecap="round" stroke-linejoin="round" d="m4.5 12.75 6 6 9-13.5" /&gt;
          &lt;/svg&gt;
        &lt;/ng-template&gt;
      &lt;/headwind-checkbox-button&gt;

      &lt;label class="cursor-pointer select-none text-sm"&gt; I like Headwind UI &lt;/label&gt;
    &lt;/headwind-checkbox&gt;
  `,
  animations: [
    trigger('scaleUp', [
      state(
        'void',
        style({
          transform: 'scale(0)',
        }),
      ),
      transition(
        'void =&gt; *',
        animate(
          '.1s',
          style({
            transform: 'scale(1)',
          }),
        ),
      ),
      transition('* =&gt; void', animate('.1s')),
    ]),
  ],
})
export class BasicCheckboxExampleComponent {}

Checkbox Classes

Each component used to implement Checkbox has the same class name as its name. You can set styles using the class name as a selector in a style file.

Checked State

You can use the checked attribute to toggle the checked state. When the checked state changes, the changed value is emitted through the checkedChange emitter.

import { Component } from '@angular/core';
import {
  HeadwindCheckboxButtonComponent,
  HeadwindCheckboxCheckedDirective,
  HeadwindCheckboxComponent,
} from '@favian/headwind-ui';

@Component({
  selector: 'app-checkbox-checked-state-example',
  standalone: true,
  imports: [HeadwindCheckboxComponent, HeadwindCheckboxButtonComponent, HeadwindCheckboxCheckedDirective],
  template: `
    &lt;headwind-checkbox [(checked)]="checked" class="inline-flex cursor-pointer select-none items-center space-x-1"&gt;
      &lt;headwind-checkbox-button
        [class]="checked ? 'border-rose-500 bg-rose-500' : 'bg-white'"
        class="relative flex h-5 w-5 items-center justify-center rounded border border-zinc-300"
      &gt;
        &lt;ng-template headwindCheckboxChecked&gt;
          &lt;svg
            xmlns="http://www.w3.org/2000/svg"
            fill="none"
            viewBox="0 0 24 24"
            stroke-width="1.5"
            stroke="currentColor"
            class="absolute h-4 w-4 stroke-white"
          &gt;
            &lt;path stroke-linecap="round" stroke-linejoin="round" d="m4.5 12.75 6 6 9-13.5" /&gt;
          &lt;/svg&gt;
        &lt;/ng-template&gt;
      &lt;/headwind-checkbox-button&gt;
    &lt;/headwind-checkbox&gt;
  `,
})
export class CheckboxCheckedStateExampleComponent {
  checked = true;
}

toggle, check, uncheck

NgModel, FormControl

The &lt;headwind-checkbox&gt; component can bind value using NgModel or FormControl.

import {
  HeadwindCheckboxButtonComponent,
  HeadwindCheckboxCheckedDirective,
  HeadwindCheckboxComponent,
} from '@favian/headwind-ui';
import { FormControl, FormsModule, ReactiveFormsModule } from '@angular/forms';

  standalone: true,
  imports: [
    HeadwindCheckboxComponent,
    HeadwindCheckboxButtonComponent,
    HeadwindCheckboxCheckedDirective,
    FormsModule,
    ReactiveFormsModule,

          class="relative flex h-5 w-5 items-center justify-center rounded border border-zinc-300 bg-white"
        &gt;

            &lt;svg
              xmlns="http://www.w3.org/2000/svg"
              fill="none"

              stroke="currentColor"

            &lt;/svg&gt;
          &lt;/ng-template&gt;

        &lt;label class="cursor-pointer select-none text-sm"&gt; With NgModel: {{ checked }} &lt;/label&gt;

Animation

Bind an animation trigger to the content of [headwindCheckboxChecked] to play animation when checked state is toggled.

import { Component } from '@angular/core';
import {
  HeadwindCheckboxButtonComponent,
  HeadwindCheckboxCheckedDirective,
  HeadwindCheckboxComponent,
} from '@favian/headwind-ui';
import { animate, state, style, transition, trigger } from '@angular/animations';

@Component({
  selector: 'app-checkbox-animation-example',
  standalone: true,
  imports: [HeadwindCheckboxComponent, HeadwindCheckboxButtonComponent, HeadwindCheckboxCheckedDirective],
  template: `
    &lt;headwind-checkbox class="inline-flex cursor-pointer select-none items-center space-x-1"&gt;
      &lt;headwind-checkbox-button
        class="relative flex h-5 w-5 items-center justify-center rounded border border-zinc-300 bg-white"
      &gt;
        &lt;ng-template headwindCheckboxChecked&gt;
          &lt;svg
            [@scaleUp]
            xmlns="http://www.w3.org/2000/svg"
            fill="none"
            viewBox="0 0 24 24"
            stroke-width="1.5"
            stroke="currentColor"
            class="absolute h-4 w-4 stroke-rose-500"
          &gt;
            &lt;path stroke-linecap="round" stroke-linejoin="round" d="m4.5 12.75 6 6 9-13.5" /&gt;
          &lt;/svg&gt;
        &lt;/ng-template&gt;
      &lt;/headwind-checkbox-button&gt;
    &lt;/headwind-checkbox&gt;
  `,
  animations: [
    trigger('scaleUp', [
      state(
        'void',
        style({
          transform: 'scale(0)',
        }),
      ),
      transition(
        'void =&gt; *',
        animate(
          '.1s',
          style({
            transform: 'scale(1)',
          }),
        ),
      ),
      transition('* =&gt; void', animate('.1s')),
    ]),
  ],
})
export class CheckboxAnimationExampleComponent {}

Accessibility

Mouse Interaction

Clicking on the &lt;headwind-checkbox&gt; element toggles its checked state.

Keyboard Interaction

Command Target Description
Space &lt;headwind-checkbox&gt; Toggle the checked state.

API

HeadwindCheckboxComponent

The root component of Checkbox.

Selector

&lt;headwind-checkbox&gt;

Host Class

.headwind-checkbox<br/> .headwind-checked when Checkbox checked.

Inputs

Name Description
@Input() set checked(value: boolean) Set the checked state of checkbox.
@Input() set disabled(value: boolean) Set the disabled state of checkbox.

Outputs

Name Description
@Output() checkedChange: EventEmitter&lt;boolean&gt; Emits the changed checked state of checkbox.

Methods

Name Description
toggle() Toggle the checked state of Checkbox.
check() Check the checkbox. If it is already checked, it is ignored.
uncheck() Uncheck the checkbox. If it is already unchecked, it is ignored.

HeadwindCheckboxButtonComponent

A component for drawing a checkbox containing a checked icon.

Selector

&lt;headwind-checkbox-button&gt;

Host Class

.headwind-checkbox-button

HeadwindCheckboxCheckedDirective

A directive for a template that wraps a checked indicator.

Selector

ng-template[headwindCheckboxChecked]
